#define RX_BUFFER_SIZE 50  //定义接收缓存大小50，可以根据需要更改

/*****************************************************************************
*  函数介绍  ： 初始化系统时钟，和运行时钟
*            ：
*    参数    ： 无
*   返回值   ： 无
******************************************************************************/
extern void setSystemCLK();

/*****************************************************************************
*  函数介绍  ： 初始化串口
*            ：
*    参数    ： 无
*   返回值   ： 无
******************************************************************************/
extern void initUart();


/*****************************************************************************
*  函数介绍  ： 发送一个字符
*            ：
*    参数    ： ch : 将要发送的字符
*   返回值   ： 无
******************************************************************************/
extern void sentChar(char ch);


/*****************************************************************************
*  函数介绍  ： 发送字符串
*            ：
*    参数    ： str : 字符串指针
*            ： len : 字符串长度
*   返回值   ： 无
******************************************************************************/
extern void sentString(char * str, int len);


/*****************************************************************************
*  函数介绍  ： 读取一个字符
*            ：
*    参数    ： 无
*   返回值   ： 无
******************************************************************************/
extern char readChar();

/*****************************************************************************
  *  函数介绍  ： 读取len个字符，如果缓存中够len个字符，则读取len个。否则，读取
  *            ： 缓存中的所有字符
  *    参数    ： str : 用于保存字符串的位置
  *            ： len : 要读取的长度
  *   返回值   ： 读到的字符个数
  ******************************************************************************/ 
extern int readString(char * str, int len);

/*****************************************************************************
  *  函数介绍  ：设置接收后的处理函数
  *            ：
  *    参数    ：inUART0_TXISR_function将要设置的函数指针
  *   返回值   ：无
  ******************************************************************************/ 
extern void setUART0_TXISR( void (*inUART0_TXISR_function)());

/*****************************************************************************
  *  函数介绍  ：设置发送后的处理函数
  *            ：
  *    参数    ：inUART0_RXISR_function将要设置的函数指针
  *   返回值   ：无
  ******************************************************************************/ 
extern void setUART0_RXISR( void (*inUART0_RXISR_function)());
